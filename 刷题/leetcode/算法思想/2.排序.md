## 冒泡排序 bubble sort

![img](../../../../../A_Lab/Note/MD_Pic/849589-20171015223238449-2146169197-16479550578903.gif)

```
思想：
    1.相邻两个元素对比，如果前面比后面大(小)交换
    2.一轮结束以后，最后一个一定是最大(小)的
    3.重复
```

```java
// 时间复杂度：O(n^2) 从头到尾撸完对比
// 空间复杂度：O(1) 交换用的空间

// 优化：
// 设置标志位，可以提前结束
// 数组局部有序，[i, length-1]一定有序了，[swap, i] swap是上次交换位置，后面没交换了 说明也是有序的
public static int[] bubbleSort(int[] array) {
    if(array.length == 0) {
        return array;
    }

    for(int i = 0; i < array.length; i++) {
        for(int j = 0; j < array.length - 1 - i; j++) { // 这一步(-1-i)就是把后面排序好的i个数给去掉
            if(array[j+1] < array[j]) {
                int tmp = array[j];
                array[j] = array[j+1];
                array[j+1] = tmp;
            }
        }
    }
    return array;
}
```

## 选择排序 Selection Sort

![img](../../../../../A_Lab/Note/MD_Pic/849589-20171015224719590-1433219824-16479550671665.gif)

```
思想：
    从序列里面找到最小(大)的，然后和起始交换
```

```java
// 时间复杂度： O(n^2) 从头比到位，不分好坏的从头比到位
// 空间复杂度： O(1) 交换的空间
public static int[] selectionSort(int[] array) {
    if(array.length == 0) {
        return array;
    }

    for(int i = 0; i < array.length; i++) {
        int minIndex = i;
        for(int j = i; j < array.length; j++) { // 循环找到i后面最小的数
            if(array[j] < array[minIndex]) {
                minIndex = j;
            }
        }
        int tmp = array[minIndex];
        array[minIndex] = array[i];
        array[i] = tmp;
    }
    return array;
}
```

## 插入排序 insertion sort

![img](../../../../../A_Lab/Note/MD_Pic/849589-20171015225645277-1151100000.gif)

```
思想：
    前面是已经排序好的序列，取后面的元素，插入到前面已经排序好的序列中
```

```java
// 时间复杂度：O(n^2)
// 空间复杂度：O(1)
public static int[] insertionSort(int[] array) {
    if(array.length == 0) {
        return array;
    }
    int current; // 现在要排序的新元素
    for(int i = 0; i < array.length - 1; i++) {
        current = array[i + 1];
        int preIndex = i;
        while(preIndex >= 0 && current < array[preIndex]) {
            array[preIndex + 1] = array[preIndex]; // 不停的后移
            preIndex--;
        }
        array[preIndex + 1] = current;
    }
    return array;
}
```

## 希尔排序(shell sort)

![img](../../../../../A_Lab/Note/MD_Pic/1192699-20180319094116040-1638766271.png)

```
思想：
    整个待排序的序列分割成若干个子序列做直接插入排序
    从最大的gap不停缩小-->序列变长。
```

```java
// 时间复杂度 O(nlogn)
// 空间复杂度 O(1)
public static int[] ShellSort(int[] array) {
    int len = array.length;
    int tmp, gap = len / 2; // gap就是操控序列大小的增量
    while(gap > 0) { 
        for(int i = gap; i < len; i++) { // 做直接插入排序
            tmp = array[i];
            int preIndex = i - gap;
            while(preIndex >= 0 && array[preIndex] > tmp) {
                // preIndex：前面还有这个分组的元素
                // 前面的元素比array[i]大的时候才会往后移动
                array[preIndex + gap] = array[preIndex]; //大的后移
                preIndex -= gap;
            }
            array[preIndex + gap] = tmp; // 直接插入的位置找到了
        }
        gap /= 2; // 增量持续缩小，序列长度增大
    }
    return array;
}
```

## 归并排序 (Merge Sort)

```
思想：
    分治法，两个有序的子序列，合并成一个有序的
    它是稳定的
```

```java
// 时间复杂度 O(nlogn)
// 空间复杂度 O(n) 创建了一个合并用的空间

/* 递归的去实现归并的过程 */
public static int[] MergeSort(int[] array) {
    if(array.length < 2) {
        // 0/1元素 哪里还需要排序呢
        return array;
    }
    int mid = array.length / 2;
    int[] left = Arrays.copyOfRange(array, 0, mid);
    int[] right = Arrays.copyOfRange(array, mid, array.length);
    return merge(MergeSort(left), MergeSort(right));
}

/* 两个数组要合并成一个数组 */
public static int[] merge(int[] left, int[] right) {
    int[] result = new int[left.length + right.length];
    for(int index = 0, i = 0, j = 0; index < result.length; index++) {
        // index: result的索引
        // i,j: left,right的索引
        if(i >= left.length) { // left遍历结束了
            result[index] = right[j++];
        } else if(j >= right.length) { // right遍历结束了
            result[index] = right[i++];
        } else if(left[i] > right[j]) {
            result[index] = right[j++];
        } else {
            result[index] = left[i++];
        }
    }
    return result;
}
```

## 快排 (quick Sort)

![img](../../../../../A_Lab/Note/MD_Pic/849589-20171015230936371-1413523412.gif)

```
思想：
    选出一个基准(pivot)，小的放基准前面，大的放基准后面
    迭代！
```

```java
// 时间复杂度 O(nlogn)
// 空间复杂度 O(logn)

public static int[] QuickSort(int[] array, int start, int end) {
    // 递归执行，所以需要制定好排序数组的index
    if(array.length < 1 || start < 0 || end >= array.length || start >end) {
        // 异常情况的排除
        return null;
    }
    // 递归！
    int smallIndex = partition(array, start, end);
    if(smallIndex > start) {
        QuickSort(array, start, smallIndex - 1);
    }
    if(smallIndex < end) {
        QuickSort(array, smallIndex + 1, end);
    }
    return array;
}

// 快排的具体实现，pivot的左右分开了
public static int partition(int[] array, int start, int end) {
    int pivot = (int) (start + Math.random() * (end - start + 1));
    int smallIndex = start - 1;
    swap(array, pivot, end); // 先把基准放到最后，方便中间index移动
    for(int i = start; i <= end; i++) {
        if(array[i] <= array[end]) {
            // smallIndex就是用来保存pivot的最终位置的，=也是为了后面换回去
            smallIndex++; 
            if(i > smallIndex) {
                swap(array, i, smallIndex);
            }
        }
    }
    return smallIndex;
}

// 交换数组内两个元素
public static void swap(int[] array, int i, int j) {
    int temp = array[i];
    array[i] = array[j];
    array[j] = temp;
}
```

## 堆排序(Heap Sort)

```
思想：
    近似完全二叉树的结构
    满足堆积的性质：子结点的键值/索引 总是小于/大于父节点

    根据性质，可以先构建堆，然后堆顶输出，调整堆，循环
```

```java
// 时间复杂度
// 空间复杂度

// 全局变量，记录数组array的长度
static int len;

/* 堆排序算法 */
public static int[] HeapSort(int[] array) {
    len = array.length;
    if(len < 1) {
        return array;
    }
    // 1.先创建堆
    buildMaxHeap(array);
    // 2.循环把堆的首位和末位交换，调整堆
    while(len > 0) {
        swap(array, 0, len - 1);
        len--;
        adjustHeap(array, 0);
    }
    return array;
}

/* 建立堆 */
public static void buildMaxHeap(int[] array) {
    for(int i = (len/2 - 1); i >= 0; i--) {
        adjustHeap(array, i);
    }
}

/* 调整堆 */
public static void adjustHeap(int[] array, int i) {
    int maxIndex = i;
    // 如果有左子树，并且左子树比父节点大，那最大指针-->左子树
    if(i * 2 < len && array[i * 2] > array[maxIndex]) {
        maxIndex = i * 2;
    }
    // 如果有右子树，并且右子树比父节点大，那最大指针-->右子树
    if(i * 2 + 1 < len && array[i * 2 + 1] > array[maxIndex]) {
        maxIndex = i * 2 + 1; // 就算先指向左边也无所谓
    }
    // 如果父节点不是最大，那就和最大交换位置
    // 递归的去调整和父节点交换
    if(maxIndex != i) {
        swap(array, maxIndex, i);
        adjustHeap(array, maxIndex);
    }
}
```

## 计数排序 (Counting Sort)

```
思想：
    针对有确定范围的整数，使用额外数组C，把目标数组里的元素给C中指定位置++

    1.找出最大的和最小的元素
    2.统计每个值i的元素出现的次数，保存
    3.从C的第一个元素开始，每一项和前一项相加
    4.反向填充数组
```

```java
// 时间复杂度: O(n+k) k个整数
// 空间复杂度: O(K)

public static int[] CountingSort(int[] array) {
    if(array.length == 0) {
        return array;
    }
    int bias, min = array[0], max = array[0];
    for(int i = 1; i < array.length; i++) {
        if(array[i] > max) {
            max = array[i];
        }
        if(array[i] < min) {
            min = array[i];
        }
    }
    bias = 0 - min;
    int[] bucket = new int[max - min + 1]; // 这就是额外的数组C了
    Array.fill(bucket, 0); // 填充新数组

    for(int i = 0; i < array.length; i++) { // 计数拉
        bucket[array[i] + bias]++;
    }
    int index = 0, i = 0;
    while(index < array.length) {
        if(bucket[i] != 0) {
            array[index] = i - bias; // bias是负数 这一点必须要记住
            bucket[i]--;
            index++;
        } else {
            i++; // 轮到下一个计数位置了
        }
    }
    return array;
}
```

## 桶排序(Bucket Sort)

![img](../../../../../A_Lab/Note/MD_Pic/849589-20171015232107090-1920702011.png)

```
思想：
    计数排序的升级版目标数据必须是要均匀的
    数据通过映射函数放到对应的桶里面，每个桶有自己的排序
    1.设置BucketSize，每个桶能放几个不一样的数值，重复的无所谓
    2.遍历原数组，一个个放到桶里面
    3.对桶排序，随便什么排序方法，也可以递归桶排序
    4.从不是空的桶里把排好序的数据拼接起来
```

```java
// 时间复杂度 O(n+k)
// 空间复杂度 O(n+k)

public static ArrayList<Integer> BucketSort(ArrayList<Integer> array, int bucketSize) {
    if(array == null || array.size() < 2) {
        // 这里的array是ArrayList类
        return array;
    }

    int max = array.get(0); min = array.get(0);
    // 找最大最小值
    for(int i = 0; i < array.size(); i++) {
        if(array.get(i) > max) {
            max = array.get(i);
        }
        if(array.get(i) < min) {
            min = array.get(i);
        }
    }

    /*----------------构建新的桶的数组结构----------------------------*/
    int bucketCount = (max - min) / bucketSize + 1;
    // bucketArr 就是new出来的桶
    ArrayList<ArrayList<Integer>> bucketArr = new ArrayList<>(bucket);
    ArrayList<Integer> resultArr = new ArrayList<>;
    // 先把桶给构建出来
    for(int i = 0; i < bucketCount; i++) {
        bucketArr.add(new ArrayList<Integer>());
    }
    /* ---------------原数组的元素塞到这个新构建的桶里面去 -------------*/
    for(int i = 0; i < array.size(); i++) {
        // 遍历原数组
        // get方法获取是那个桶，add把对应的数值加上去
        bucketArr.get((array.get(i) - min) / bucketSize).add(array.get(i));
    }

    /*---------------遍历桶输出排序好的元素了-------------------------*/
    for(int i = 0; i < bucketCount; i++) {
        if(bucketSize == 1) { // 这就是计数排序！
            for(int j = 0; j < bucketArr.get(i).size(); j++) {
                resultArr.add(bucketArr.get(i).get(j));
            }
        } else {
            if(bucketCount == 1) { // 从最后一列向前走的意思
                bucketSize--;
            }
            ArrayList<Integer> temp = BucketSort(bucketArr.get(i), bucketSize); // 递归桶排序
            for(int j = 0; j < temp.size(); j++) {
                result.add(temp.get(j));
            }
        }
    }
    return resultArr;
}
```

## 基数排序(Radix Sort)

![img](../../../../../A_Lab/Note/MD_Pic/849589-20171015232453668-1397662527.gif)

```
思想：
    按照低位排序，然后收集，再往高位推
    1.取得数组里最大的数，取得位数
    2.arr作为原始数组，最低位开始取每个位组成radix数组
    3.对radix进行计数排序
```

```java
// 时间复杂度
// 空间复杂度

public static int[] RadixSort(int[] array) {
    if(array == null || array.length < 2) {
        return array;
    }

    // 1.最大数的位数
    int max = array[0];
    for(int i = 1; i < array.length; i++) {
        max = Math.max(max, array[i]);
    }
    // 取得位数
    int maxDigit = 0;
    while(max != 0) {
        max /= 10;
        maxDigit++;
    }
    // new出结构
    int mod = 10, div = 1;
    ArrayList<ArrayList<Integer>> bucketList = new ArrayList<>();
    for(int i = 0; i < 10; i++) {
        bucketList.add(new ArrayList<Integer>());
    }

    for(int i = 0; i < maxDigit; i++, mod *= 10, div *= 10) {
        // 遍历的同时根据i的位数来存储array分组
        for(int j = 0; j < array.length; j++) {
            int num = (array[j] % mod) / div;
            bucketList.get(num).add(array[j]);
        } // 粗排序一次就结束了

        int index = 0;
        for(int j = 0; j < bucketList.size(); j++) {
            for(int k = 0; k < bucketList.get(j).size(); k++) {
                array[index++] = bucketList.get(j).get(k);
            }
            bucketList.get(j).clear();
        }
        // array更新了一次
    }
    return array;
}
```
